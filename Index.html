Sample SQL Query


-- Total Revenue from Pizza Sales
SELECT ROUND(SUM(order_details.quantity * pizzas.price), 2) AS total_pizza_sales
FROM order_details
JOIN pizzas ON pizzas.pizza_id = order_details.pizza_id;
